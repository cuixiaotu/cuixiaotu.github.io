排序大全

排序是将一组对象按照某种逻辑顺序重新排列的过程。
每次都草草的看下排序似懂非懂，知其然不知其所以然,这次用go简单的实现一下。

1.冒泡排序

思路:两两比较 轻的上浮 重的下沉

func Bubble(comparable []int) {
    n := len(comparable)
    for i := 0; i < n; i++ { //n次是必须的
        //优化思路 每次比较完 最重的已经沉到底了（多减1 是两两比较的原因）
        for j := 0; j < n-i-1; j++ {
            if less(comparable[j+1], comparable[j]) {
                exchange(comparable, j, j+1)
            }
        }
    }
}

2.选择排序

思路:每次选出最小值放在将它和数组的第一个元素交换位置，再在剩余的元素中反复操作

func Selection(comparable []int) {
    n := len(comparable)
    for i := 0; i < n; i++ { //进行N次变换
        min := i //每次重置最小值为开始值
        for j := i + 1; j < n; j++ {
            if less(comparable[j], comparable[min]) {
                min = j
            }
        }
        exchange(comparable, i, min)
    }
}

3.插入排序

思路：按照理牌的思路 每来一张牌 把牌放在已经排序好的合适位置

func Insertion(comparable []int) {
    n := len(comparable)
    for i := 1; i < n; i++ {
        k := i //拿到第i张牌的时候
        for {//嘿嘿 这里用的算法导论里的while 大家用for一个原理
            //只有一张牌|当前牌重置到第一张|当前牌到了合适位置 从右向左移动
            if k <= 0 || less(comparable[k-1], comparable[k]) {
                break
            }
            exchange(comparable, k, k-1)
            k-- //当然 每次交换位置后 k减1
        }
    }
}

4.希尔排序

思路:插入排序的优化 前面的排序只会交换相邻位置 希尔交换不相邻元素对数组局部进行排序
使数组中间隔为h的元素都是有序的 h按一定规律（未有最优递增序列）递减

func Shell(comparable []int) {
    n := len(comparable)
    h := 1
    //设置步长 (递增序列)
    for h < n/3 {
        h = 3*h + 1
    }

    for h >= 1 { //进行子数组排序  h为1时为插入排序
        for i := h; i < n; i++ {
            k := i //拿到第i张牌的时候
            for {
                //少于k张牌|当前牌重置到第一张|当前牌到了合适位置 从右向左移动
                if k < h || less(comparable[k-h], comparable[k]) {
                    break
                }
                exchange(comparable, k, k-h)
                k -= h //当然 每次交换位置后 k减h
            }
        }
        h = h / 3
    }
}

5.归并排序

思路：分而治之

var aux []int
func Merge(comparable []int) {
    //直接多申请一份数组空间 避免归并过程中反复申请回收
    aux = make([]int, len(comparable),cap(comparable))
    //topDownMerge(comparable, 0, len(comparable) - 1)
    bottomUpMerge(comparable, 0, len(comparable) - 1)
}

//自顶向下归并
func topDownMerge(comparable []int, lo , hi int) {
    if hi <= lo {
        return
    }
    mid := lo + (hi-lo)/2
    topDownMerge(comparable,lo,mid)
    topDownMerge(comparable,mid+1,hi)
    merge(comparable, lo, mid, hi)
}

//自顶向下归并
func bottomUpMerge(comparable []int) {
    n := len(comparable)
    for sz := 1; sz < n; sz += sz { //sz子数组的大小
        for lo := 0; lo < n-sz; lo += sz + sz { //看做sz为1的子数组 向上合并的过程
            merge(comparable, lo, lo+sz-1, int(math.Min(float64(lo+sz+sz-1), float64(n-1))))
        }
    }
}

//原地归并(归并的重点 合并两个有序数组 并排好序)
//将a[lo,mid] 和 a[mid+1,hi]归并
func merge(a []int, lo , mid , hi int) {
    //两个数组的首位比较
    i , j := lo , mid+1

    //临时数组 直接赋值会引用同一个空间 导致异常
    for k := lo; k <= hi; k++ {
        aux[k] = a[k]
    }

    for k := lo; k <= hi; k++ {
        if i > mid {    //边界处理 左侧首元素大于尾元素位数时 表示左侧数组已经用尽
            a[k] = aux[j]
            j++
        }else if j > hi {//边界处理 右侧首元素大于尾元素位数时 表示右侧数组已经用尽
            a[k] = aux[i]
            i++
        } else if less(aux[j], aux[i]) {//右侧表左侧小 取右侧呀 mark 必须是原数据(复制出来的aux)啊 写错了排查了半天
            a[k] = aux[j]
            j++
        } else {//正常取左侧
            a[k] = aux[i]
            i++
        }
    }
}

6.快速排序

思路：将一个数组分为两个子数组 ，将两部分独立的排序。
将随机挑选的元素放在合适的位置（老数组排序好的位置），然后在递归调用排序前后两个子数组

优化思路

1).小数组改为插入排序
2).三取样切分(使用子数组的一小部分元素的中位数切分数组 切分的好 效率肯定高 也要考虑代价 计算中位数)
3).熵最优排序 (对于重复元素的优化 将数组分为大于 小于 等于切分元素的三部分数组)

func Quick(comparable []int) {
    quickSort(comparable, 0, len(comparable)-1)
}

func quickSort(comparable []int, lo, hi int) {
    if hi <= lo {
        return
    }
    j := partition(comparable, lo, hi)
    quickSort(comparable, lo, j-1)
    quickSort(comparable, j+1, hi)
}

//快排的分区
func partition(comparable []int, lo, hi int) int {
    //将数组分为 a[lo,i-1]  a[i] a[i+1,hi]
    v := comparable[lo] //这里也可以随机挑选一个
    i := lo         //从第二个元素开始对比
    j := hi

    for {
        //第二个开始
        i++
        for less(comparable[i], v) {
            i++
            if i == hi {
                break
            }
        }

        for less(v, comparable[j]) {
            j--
            if j <= lo {
                break
            }
        }

        if i >= j {
            break
        }

        exchange(comparable, i, j)
    }
    //把v=a[j]放到正确的位置
    exchange(comparable, lo, j)
    return j
}

7.堆排序

思路：基于优先队列的排序方法 将所有元素插入一个查找最小元素的优先队列
然后再重复调用删除最小元素的操作 将他们顺序删除
二叉堆的表现方式：只用数组 不用指针 也就是树的顺序表达法
从a[0]开始的话 子节点在父节点（n位）的2n+1和2n+2位置上

func Heap(comparable []int) {

    n := len(comparable)
    //构建堆
    for k := (n - 1) / 2; k >= 0; k-- {
        sink(comparable, k, n)
    }

    for i := n - 1; i > 0; i-- {
        exchange(comparable, 0, i)
        sink(comparable, 0, i)
    }
}


//堆的下沉方法
func sink(comparable []int, k, n int) {
    j := 2*k + 1
    for j < n {
        if j+1 < n && less(comparable[j+1], comparable[j]) {
            j++
        }
        if !less(comparable[j], comparable[k]) {
            break
        }
        exchange(comparable, k, j)
        k = j
        j = 2*k + 1
    }
}

8.其他方法

//比较数值大小
func less(a, b int) bool {
    v := true
    if a >= b {
        v = false
    }
    return v
}

//换位
func exchange(comparable []int, i , j int) {
    comparable[i], comparable[j] = comparable[j], comparable[i]
}

//是否排完序
func isSorted(comparable []int) bool {
    for i := 1; i < len(comparable); i++ {
        if less(comparable[i], comparable[i-1]) {
            return false
        }
    }
    return true
}

1.docker安装

1.1 官方仓库安装

# 官方推荐使用官方仓库安装 方便安装和更新docker
# 更新apt包  允许apt通过https使用stable仓库 
# 查看内核知道安装哪个版本
uname -a
# Linux cuixiaotu 5.4.0-77-generic #86-Ubuntu SMP Thu Jun 17 02:35:03 UTC 2021 x86_64 x86_64 x86_64 GNU/Linux

cat /proc/version
# Linux version 5.4.0-77-generic (buildd@lgw01-amd64-028) (gcc version 9.3.0 (Ubuntu 9.3.0-17ubuntu1~20.04)) #86-Ubuntu SMP Thu Jun 17 02:35:03 UTC 2021



sudo apt-get update

sudo apt-get install \
    apt-transport-https \
    ca-certificates \
    curl \
    gnupg \
    lsb-release
    
# 添加Docker官方GPG密钥    
curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo gpg --dearmor -o /usr/share/keyrings/docker-archive-keyring.gpg

# 获取当前操作系统代号
lsb_release -cs 
#focal

# 以下命令设置stable仓库。 
 echo \
  "deb [arch=amd64 signed-by=/usr/share/keyrings/docker-archive-keyring.gpg] https://download.docker.com/linux/ubuntu \
  $(lsb_release -cs) stable" | sudo tee /etc/apt/sources.list.d/docker.list > /dev/null
  

sudo apt-get update
sudo apt-get install docker-ce docker-ce-cli containerd.io

2. 安装后的配置

Docker 守护进程绑定到 Unix 套接字而不是 TCP 端口。默认情况下，Unix 套接字由用户 root 拥有，其他用户只能使用 sudo 访问它。 Docker 守护进程始终以 root 用户身份运行。

如果不想在 docker 命令前加上 sudo，请创建一个名为 docker 的 Unix 组并向其中添加用户。当 Docker 守护进程启动时，它会创建一个可由 docker 组成员访问的 Unix 套接字

2.1 配置Docker服务

为了避免每次使用Docker都需要切换身份，可以把当前用户添加到安装创建docker用户组

sudo groupadd docker

# 由于linux用户权限是再用户登录的时候加载的  分配权限后需要退出再登录
sudo usermod -aG docker $USER

exit

docker -v
#Docker version 20.10.7, build f0df350

docker run hello-wored

2.2 配置镜像加速器

这里使用阿里云镜像加速器，登录阿里云-容器镜像服务-镜像加速器可获取加速地址

sudo mkdir -p /etc/docker
sudo tee /etc/docker/daemon.json <<-'EOF'
{
  "registry-mirrors": ["https://xxxxxx.mirror.aliyuncs.com"]
}
EOF
sudo systemctl daemon-reload
sudo systemctl restart docker
